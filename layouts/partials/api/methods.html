{{/* Methods expect a context with: - .reference: a reference to the whole api spec - .path: the path of the endpoint these methods belong to - .methods: the methods that belong to the path */}} {{- $reference := .reference -}} {{- $path := .path -}}
{{- range $method, $details := .methods -}} {{- $context := dict "reference" $reference "path" $path "method" $method "details" $details -}} {{- partial "api/method.html" $context -}} {{- end -}}
